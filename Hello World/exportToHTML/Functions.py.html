# make a quiz game using functions
# validate the question with function
# or use the whole thing as a function.


print("welcome to my quiz!")
score = 0

def quiz_question(question, answer, score):
    user_answer = input(question)

    try:
        user_answer = int(user_answer)
    except ValueError:
        user_answer = user_answer.strip().lower()

    if user_answer == answer:
        print("you got it!")
        score += 1
    else:
        print("wrong one!")

    print("Score is "+str(score))
    return score


score = quiz_question("How many days in a week?: ", 7, score)

score = quiz_question("How many states in USA?: ", 50, score)

score = quiz_question("What is the opposite of day? ", "night", score)

print("Final score is " + str(score))
